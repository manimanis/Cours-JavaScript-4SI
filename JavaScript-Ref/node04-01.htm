String

Fonction charAt(...)

Syntaxe

chaine.charAt(pos)

Exemple

<script type="text/javascript">

var str = "Hello world!";
var m = parseInt(str.length / 2);
document.write("Premier caractère : " + str.charAt(0) + "<br />");
document.write("Caractère du milieu : " + str.charAt(m) + "<br />");
document.write("Dernier caractère : " + str.charAt(str.length-1));

</script>
