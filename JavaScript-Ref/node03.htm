Number

Fonction toFixed(...)

Syntaxe

value.toFixed()
value.toFixed(x)

Exemple

<script type="text/javascript">

var num = new Number(13.3714);
document.write(num.toFixed()+"<br />");
document.write(num.toFixed(1)+"<br />");
document.write(num.toFixed(3)+"<br />");
document.write(num.toFixed(10));

</script>
